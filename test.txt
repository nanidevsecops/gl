apiVersion: v1
kind: Pod
metadata:
    name: advanced-scheduling-pod
spec:
    containers:
    - name: my-container
        image: nginx
    affinity:
        nodeAffinity:
            requiredDuringSchedulingIgnoredDuringExecution:
                nodeSelectorTerms:
                - matchExpressions:
                    - key: disktype
                        operator: In
                        values:
                        - ssd
            preferredDuringSchedulingIgnoredDuringExecution:
            - weight: 1
                preference:
                    matchExpressions:
                    - key: region
                        operator: In
                        values:
                        - us-west
    tolerations:
    - key: "key1"
        operator: "Equal"
        value: "value1"
        effect: "NoSchedule"
